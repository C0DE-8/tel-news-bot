# Local API test examples
#
# Start the server first:
# cd backend
# npm start
#
# Or with nodemon:
# npm run dev
#
# Default local URL:
BASE_URL=http://localhost:3000
ADMIN_HEADER="x-admin-chat-id: ADMIN_USER_ID"

# Protected HTTP routes require x-admin-chat-id to match TELEGRAM_ADMIN_CHAT_IDS.
# Telegram itself does not send this header to POST /webhook/telegram.

### Health
curl -s http://localhost:3000/health

### Bot runtime status
curl -s http://localhost:3000/bot/status \
  -H "x-admin-chat-id: ADMIN_USER_ID"

### Test router ping
curl -s http://localhost:3000/test/ping \
  -H "x-admin-chat-id: ADMIN_USER_ID"

### Admin: list every group config
curl -s http://localhost:3000/admin/news-config \
  -H "x-admin-chat-id: ADMIN_USER_ID"

### Admin: list known groups the bot has seen
curl -s http://localhost:3000/admin/groups \
  -H "x-admin-chat-id: ADMIN_USER_ID"

### Admin: configure a group to post crypto news every 30 minutes
### Replace CHAT_ID with your Telegram group chat id.
curl -s -X POST http://localhost:3000/admin/news-config \
  -H "Content-Type: application/json" \
  -H "x-admin-chat-id: ADMIN_USER_ID" \
  -d '{
    "chatId": "CHAT_ID",
    "topic": "crypto",
    "intervalMinutes": 30,
    "postLimit": 10,
    "postNow": false
  }'

### Admin: configure politics news to start at a specific time
### postAt accepts any valid ISO date string.
curl -s -X POST http://localhost:3000/admin/news-config \
  -H "Content-Type: application/json" \
  -H "x-admin-chat-id: ADMIN_USER_ID" \
  -d '{
    "chatId": "CHAT_ID",
    "topic": "politics",
    "intervalMinutes": 60,
    "postLimit": 5,
    "postAt": "2099-01-01T18:00:00.000Z"
  }'

### Admin: post one item immediately after saving config
curl -s -X POST http://localhost:3000/admin/news-config \
  -H "Content-Type: application/json" \
  -H "x-admin-chat-id: ADMIN_USER_ID" \
  -d '{
    "chatId": "CHAT_ID",
    "topic": "crypto",
    "intervalMinutes": 30,
    "postLimit": 3,
    "postNow": true
  }'

### Admin: stop news for a group
curl -s -X POST http://localhost:3000/admin/news-stop \
  -H "Content-Type: application/json" \
  -H "x-admin-chat-id: ADMIN_USER_ID" \
  -d '{"chatId":"CHAT_ID"}'

### Webhook route: Telegram will POST this shape to your server
curl -s -X POST http://localhost:3000/webhook/telegram \
  -H "Content-Type: application/json" \
  -d '{
    "update_id": 1001,
    "message": {
      "message_id": 2,
      "chat": { "id": "CHAT_ID", "type": "group" },
      "text": "/setnews crypto"
    }
  }'

### API route: set Telegram webhook
### Put your public HTTPS base URL in WEBHOOK_URL.
curl -s -X POST http://localhost:3000/webhook/telegram/set \
  -H "Content-Type: application/json" \
  -H "x-admin-chat-id: ADMIN_USER_ID" \
  -d '{
    "url": "WEBHOOK_URL/webhook/telegram",
    "dropPendingUpdates": false
  }'

### API route: Telegram webhook info
curl -s http://localhost:3000/webhook/telegram/info \
  -H "x-admin-chat-id: ADMIN_USER_ID"

### API route: delete Telegram webhook if you want to go back to polling
curl -s -X DELETE http://localhost:3000/webhook/telegram \
  -H "x-admin-chat-id: ADMIN_USER_ID"
